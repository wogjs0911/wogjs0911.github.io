package ex1.test;

public class OmokTest2 {

	public static void main(String[] args) {

					System.out.printf("%c", '○');	
				else if (y==1)									// 코드는 수정하기에 편해야 한다.(유지보수)
					System.out.printf("%c", '┬');			
				else if (y==10)
					System.out.printf("%c", '┴');			// 이 코드는 고치기 어렵다.

					System.out.printf("%c", '├');			
				else if (x==10)
					System.out.printf("%c", '┤');

				else 
					System.out.printf("%c", '┼');

		}
		
	}

}
