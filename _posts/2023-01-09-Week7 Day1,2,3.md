        }


        // ------------------- 이동을 위한 로직 ------------
    }

    moveTo(dx,dy){
        
        let w = dx - this.x;                // 목적지 위치 - 현재 위치(업데이트되고나서) = 남은 위치 거리 계산 
        let h = dy - this.y;

        this.vy = h / d;
        
        this.dx = dx;
        this.dy = dy;
    }
