문제 3. 1)

```java


package com.test.monthly1_1;


import java.util.Scanner;

public class MontlyTest1_1 {

	public static void main(String[] args) {
	
		
		System.out.printf("평균 : %.2f",average());
		
	}

	private static float average() {
		
		Scanner scan = new Scanner(System.in);
		System.out.println("3개의 값을 입력해주세요.");
		
		int[] nums = new int[3];
		int sum = 0;
		
		
		for(int i =0; i<3; i++) {
			nums[i] = scan.nextInt();
			sum += nums[i];
		}
		
		
		return (float) (sum/3.0);
		
	}

}

```

package com.test.monthly1_2;

import java.util.Scanner;

public class MontlyTest1_2 {

	public static void main(String[] args) {
	
		
		System.out.printf("평균 : %.2f", average());
		
	}

	private static float average() {
		
		Scanner scan = new Scanner(System.in);
		System.out.println("값을 입력해주세요.");
		
		int[] nums = {0};
		int i = 0;
		int sum = 0;
		
		while(scan.hasNext()) {			// 그냥 length 이용
			nums[i++] = scan.nextInt();
			sum += nums[i++];
		}
		
		System.out.println(sum);
		
		
		return (float) (sum / 3.0);
		
	}

}

package com.test.monthly1_3;

import java.util.Random;

public class Lotto {

	int[] lottoNums = new int[6];
	

	public Lotto() {
		
	}
	
	public void generateNumbers() {
		
		Random rand = new Random();
		
		for(int i=0; i<6; i++) {
			lottoNums[i] = rand.nextInt(49) + 1;	 // 별찍기 이용하기(i와 j+1 범위 / 5-j 범위) 			
			System.out.println(lottoNums[i]);
		}
	}
}
